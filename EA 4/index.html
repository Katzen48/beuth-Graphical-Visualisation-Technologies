$$ x(r,\alpha) = \cos(\alpha) * (1 + \frac{r}{2}\cos\frac{\alpha}{2}) $$

$$ y(r,\alpha) = \sin(\alpha) * (1 + \frac{r}{2}\cos\frac{\alpha}{2}) $$

$$ z(r,\alpha) = \frac{r}{2}\sin\frac{\alpha}{2} $$

$$ r = \{-1,...,1\} $$

$$ x = u\cos(v) $$

$$ y = u\sin(v) $$

$$ z = \log(u) $$

$$ u = \{0.1,...,2\} $$

$$ v = \{0,...,2\pi\} $$
